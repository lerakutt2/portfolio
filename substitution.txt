#include <cs50.h>
#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main(int argc, string argv[])
{
    if (argc != 2 || strlen(argv[1]) != 26)
    {
        printf("иди в жопу\n");
        return 1;
    }
    string key = argv[1];
    for (int i = 0; i < 26; i++)
    {
        if (key[i] < 65 || key[i] > 122 || (key[i] > 90 && key[i] < 97))
        {
            printf("Выйди и зайди нормально!\n");
            return 1;
        }
        else if ((key[i] >= 65 && key[i] <= 90) || (key[i] >= 97 && key[i] <= 122))
        {
            for (int j = 0; j < 26; j++)
            {
                if ((toupper(key[i]) == toupper(key[j])) && (i != j))
                {
                    printf("Повторяться нельзя, придурок\n");
                    return 1;
                }
            }
        }
    }
    string text = get_string("plaintext: ");
    for (int i = 0; i < strlen(text); i++)
    {
        if (text[i] >= 65 && text[i] <= 90)
        {
            text[i] = toupper(key[text[i] - 65]);
        }
        else if (text[i] >= 97 && text[i] <= 122)
        {
            text[i] = tolower(key[text[i] - 97]);
        }
    }
    printf("ciphertext: %s\n", text);
}
