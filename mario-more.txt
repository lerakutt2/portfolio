#include <cs50.h>
#include <stdio.h>

int main(void)
{
    int h;
    // Prompt user to type number from 1 to 8
    do
    {
        h = get_int("Height: ");
    }
    while (h <= 0);
    for (int i = 0; i < h; i++)
    {
        for (int j = 0; j < h - i - 1; j++)
        {
            printf(" ");
        }
        // This loop types height-i-1 number of spaces
        for (int j = 0; j < i + 1; j++)
        {
            printf("#");
        }
        // This loop types left section of #
        printf("  ");
        for (int j = 0; j < i + 1; j++)
        {
            printf("#");
        }
        // This loop types right section of #
        printf("\n");
        // Going to the next line of code
    }
}
