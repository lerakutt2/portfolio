string but(int x)
{
    int y = x % 10;
    if (y >= 5 || y == 0)
        return "бутылок";
    else if (!(x <= 14 && x >= 11) && y <= 4 && y >= 2)
        return "бутылки";
    else if (y == 1)
        return "бутылка";
    else
        return "бутылок";
}
